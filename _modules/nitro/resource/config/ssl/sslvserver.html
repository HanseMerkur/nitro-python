    def get(cls, client, name="", option_="") :
        """Use this API to fetch all the sslvserver resources that are configured on netscaler.

        :param client: 
        :param name:  (Default value = "")
        :param option_:  (Default value = "")

        """

            if not name :
                obj = sslvserver()
                response = obj.get_resources(client, option_)
            else :
                if type(name) != cls :
                    if type(name) is not list :
                        obj = sslvserver()
                        obj.vservername = name
                        response = obj.get_resource(client, option_)
                    else :
                        if name and len(name) > 0 :
                            response = [sslvserver() for _ in range(len(name))]
                            obj = [sslvserver() for _ in range(len(name))]
                            for i in range(len(name)) :
                                obj[i] = sslvserver()
                                obj[i].vservername = name[i]
                                response[i] = obj[i].get_resource(client, option_)
            return response
        except Exception as e :
            raise e
