        try :
            if not name :

                if type(name) != cls :
                    if type(name) is not list :
                        obj = responderpolicy()
                        obj.name = name

                        if name and len(name) > 0 :
                            response = [responderpolicy() for _ in range(len(name))]
                            obj = [responderpolicy() for _ in range(len(name))]
                            for i in range(len(name)) :
